// 1. 原始名單與競賽順序 (作備援使用) const playersList = [ "小菜", "Felix", "王煒強", "Mu", "漢克", "豪以", "Bo-Yi", "豬餅", "小白兔", "陀螺屁眼", "Aaron", "深淵爆轉猴", "泡麵", "林北尬周", "伊森", "華起", "小陳", "要不要吃陀螺燒", "牛魔王", "紅豆年糕", "勁廷", "阿良", "捲爸", "雷克斯", "夫夫", "小峰", "酷狗壞小子", "阿杜", "Eric", "南無阿彌陀螺", "小卡比", "大肌霸" ]; // 2. 初始化賽事數據結構 let tournamentData = { r32: [], r16: [] }; // ── 【核心改動 1：讀取本地存檔】 ── function loadTournamentData() { const savedData = localStorage.getItem("gyro_tournament_save"); if (savedData) { try { // 如果有存檔，直接解開 JSON 使用 tournamentData = JSON.parse(savedData); console.log("成功載入歷史賽程存檔！"); } catch (e) { console.error("存檔解析出錯，重新初始化名單", e); initNewTournament(); } } else { // 沒有存檔，初始化全新比賽 initNewTournament(); } } // 初始化全新比賽的邏輯 function initNewTournament() { tournamentData.r32 = []; tournamentData.r16 = []; // 初始化 32 強 for(let i = 0; i < 16; i++) { tournamentData.r32.push({ id: i + 1, stage: 'r32', stageName: '32強賽', red: playersList[i * 2], blue: playersList[i * 2 + 1], redScore: 0, blueScore: 0, winner: null }); } // 初始化 16 強 for(let i = 0; i < 8; i++) { tournamentData.r16.push({ id: i + 1, stage: 'r16', stageName: '16強賽', red: `32強 M${i*2+1} 勝者`, blue: `32強 M${i*2+2} 勝者`, redScore: 0, blueScore: 0, winner: null, sourceMatchIds: [i*2+1, i*2+2] }); } } // ── 【核心改動 2：儲存資料到本地】 ── function saveTournamentData() { // 將物件轉換為字串存入 localStorage localStorage.setItem("gyro_tournament_save", JSON.stringify(tournamentData)); } // 全局當前選中比賽 let currentStage = 'r32'; let currentMatchId = 1; // 3. 頁面初始化 document.addEventListener("DOMContentLoaded", () => { loadTournamentData(); // 先讀取存檔 renderSelectOptions(); renderR32Grid(); renderR16Grid(); loadMatch(currentStage, currentMatchId); }); // 切換頁面分頁 function switchPage(pageId, btn) { document.querySelectorAll('.page').forEach(p => p.classList.remove('active')); document.getElementById(`page-${pageId}`).classList.add('active'); document.querySelectorAll('.nav-btn').forEach(b => { b.classList.remove('active-red', 'active-blue'); }); if(pageId === 'live') { btn.classList.add('active-red'); } else if (pageId === 'r32') { btn.classList.add('active-blue'); } else { btn.classList.add('active-red'); } } // 生成下拉選單 function renderSelectOptions() { const select = document.getElementById("match-select"); select.innerHTML = ""; let optGroup32 = document.createElement("optgroup"); optGroup32.label = "【32強淘汰賽】"; tournamentData.r32.forEach(m => { let opt = document.createElement("option"); opt.value = `r32-${m.id}`; opt.innerText = `第 ${m.id} 場：${m.red} VS ${m.blue}`; if(m.winner) opt.innerText += " (已完賽)"; optGroup32.appendChild(opt); }); select.appendChild(optGroup32); let optGroup16 = document.createElement("optgroup"); optGroup16.label = "【16強晉級賽】"; tournamentData.r16.forEach(m => { let opt = document.createElement("option"); opt.value = `r16-${m.id}`; opt.innerText = `16強 M${m.id}：${m.red} VS ${m.blue}`; if(m.winner) opt.innerText += " (已完賽)"; optGroup16.appendChild(opt); }); select.appendChild(optGroup16); } // 載入選定比賽到計分板 function loadSelectedMatch() { const val = document.getElementById("match-select").value; const parts = val.split('-'); currentStage = parts[0]; currentMatchId = parseInt(parts[1]); loadMatch(currentStage, currentMatchId); } function loadMatch(stage, id) { currentStage = stage; currentMatchId = id; const match = tournamentData[stage].find(m => m.id === id); document.getElementById("live-round-title").innerText = stage === 'r32' ? "ROUND OF 32" : "ROUND OF 16"; document.getElementById("live-match-id").innerText = `MATCH ${id}`; document.getElementById("live-red-name").innerText = match.red; document.getElementById("live-blue-name").innerText = match.blue; document.getElementById("live-red-score").innerText = match.redScore; document.getElementById("live-blue-score").innerText = match.blueScore; document.getElementById("match-select").value = `${stage}-${id}`; updateWinFlash(match); } // 分數調整與 Bump 動效 function adjustScore(side, amount) { const scoreEl = document.getElementById(`live-${side}-score`); let currentScore = parseInt(scoreEl.innerText); currentScore += amount; if(currentScore < 0) currentScore = 0; scoreEl.innerText = currentScore; scoreEl.classList.remove('bump'); void scoreEl.offsetWidth; scoreEl.classList.add('bump'); const rScore = parseInt(document.getElementById("live-red-score").innerText); const bScore = parseInt(document.getElementById("live-blue-score").innerText); const redFlash = document.getElementById("live-red-flash"); const blueFlash = document.getElementById("live-blue-flash"); redFlash.classList.remove('show'); blueFlash.classList.remove('show'); if(rScore > bScore) { redFlash.classList.add('show'); redFlash.innerText = "領先中"; } else if (bScore > rScore) { blueFlash.classList.add('show'); blueFlash.innerText = "領先中"; } } function updateWinFlash(match) { const redFlash = document.getElementById("live-red-flash"); const blueFlash = document.getElementById("live-blue-flash"); redFlash.classList.remove('show'); blueFlash.classList.remove('show'); if(match.winner === 'red') { redFlash.classList.add('show'); redFlash.innerText = "WINNER"; } else if(match.winner === 'blue') { blueFlash.classList.add('show'); blueFlash.innerText = "WINNER"; } } // 重設當前對戰 function resetCurrentMatch() { if(!confirm("確定要重設本場分數嗎？")) return; const match = tournamentData[currentStage].find(m => m.id === currentMatchId); match.redScore = 0; match.blueScore = 0; match.winner = null; if(currentStage === 'r32') { syncToR16(currentMatchId, `32強 M${currentMatchId} 勝者`); } saveTournamentData(); // 【變更後存檔】 loadMatch(currentStage, currentMatchId); renderR32Grid(); renderR16Grid(); renderSelectOptions(); } // 確認勝出，儲存並自動計算晉級 function saveAndAdvance() { const match = tournamentData[currentStage].find(m => m.id === currentMatchId); const rScore = parseInt(document.getElementById("live-red-score").innerText); const bScore = parseInt(document.getElementById("live-blue-score").innerText); if(rScore === bScore) { alert("平手無法決定勝負！請分出勝負後再點擊確認。"); return; } match.redScore = rScore; match.blueScore = bScore; match.winner = rScore > bScore ? 'red' : 'blue'; const winnerName = match.winner === 'red' ? match.red : match.blue; if(currentStage === 'r32') { syncToR16(currentMatchId, winnerName); } saveTournamentData(); // 【變更後存檔】 alert(`本場結束！【${winnerName}】成功晉級並已自動存檔！`); updateWinFlash(match); renderR32Grid(); renderR16Grid(); renderSelectOptions(); if(currentMatchId < tournamentData[currentStage].length) { loadMatch(currentStage, currentMatchId + 1); } } function syncToR16(r32MatchId, winnerName) { const r16Match = tournamentData.r16.find(m => m.sourceMatchIds.includes(r32MatchId)); if(r16Match) { if(r16Match.sourceMatchIds[0] === r32MatchId) { r16Match.red = winnerName; } else { r16Match.blue = winnerName; } } } function selectMatchFromGrid(stage, id) { switchPage('live', document.querySelectorAll('.nav-btn')[0]); loadMatch(stage, id); } function renderR32Grid() { const grid = document.getElementById("grid-r32"); grid.innerHTML = ""; tournamentData.r32.forEach(m => { const isLive = (currentStage === 'r32' && currentMatchId === m.id); const card = document.createElement("div"); card.className = `bcard ${isLive ? 'is-live' : ''}`; card.onclick = () => selectMatchFromGrid('r32', m.id); const wRed = m.winner === 'red' ? 'winner' : ''; const wBlue = m.winner === 'blue' ? 'winner' : ''; card.innerHTML = `
MATCH ${m.id} ${isLive ? 'LIVE 控制中' : (m.winner ? '已結束' : '等待中')}
${m.red} ${m.redScore}
${m.blue} ${m.blueScore}
`; grid.appendChild(card); }); } function renderR16Grid() { const grid = document.getElementById("grid-r16"); grid.innerHTML = ""; tournamentData.r16.forEach(m => { const isLive = (currentStage === 'r16' && currentMatchId === m.id); const card = document.createElement("div"); card.className = `bcard ${isLive ? 'is-live' : ''}`; card.onclick = () => selectMatchFromGrid('r16', m.id); const wRed = m.winner === 'red' ? 'winner' : ''; const wBlue = m.winner === 'blue' ? 'winner' : ''; card.innerHTML = `
16強 MATCH ${m.id} ${isLive ? 'LIVE 控制中' : (m.winner ? '已結束' : '對戰組合')}
${m.red} ${m.redScore}
${m.blue} ${m.blueScore}
`; grid.appendChild(card); }); } // 額外附贈：如果未來有新的一屆比賽想徹底重洗掉存檔，可在瀏覽器 Console 輸入「clearTournament()」 function clearTournament() { if(confirm("確定要完全清除歷史存檔，重開全新一屆比賽嗎？")) { localStorage.removeItem("gyro_tournament_save"); location.reload(); } }
